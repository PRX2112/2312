    public function regCust(Request $req)
    {
        # code..
        $file=$req->filep;
        $name=$file->getClientOriginalName();
        $ext=$file->getClientOriginalExtension();
        $basename=substr($name,0,stripos($name,'.'));
        $newfilename=md5($basename).time().rand(5,1000).'.'.$ext;
        echo $newfilename;
        $req->filep->move(public_path('uploads'),$newfilename);
        DB::table('cust_reg')->insert([
            'name'=>$req->name,
            'password'=>$req->password,
            'age'=>$req->age,
            'contact'=>$req->contact,
            'photo'=>$newfilename 
        ]);
        return Back()->with('reg_success','Customer has been registered');
    }
